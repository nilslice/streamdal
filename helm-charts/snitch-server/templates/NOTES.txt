1. To access the Snitch Console:

{{- if .Values.consoleIngress.enabled }}
NOTE: It may take a few minutes for the Ingress to be available.
You can watch the status by running 'kubectl get ingress --namespace {{ .Release.Namespace }} -w {{ include "snitch-server.fullname" . }}-console'
{{- else }}
export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "snitch-server.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
echo "Forwarding port to access Snitch Console on http://127.0.0.1:3000/"
kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 3000:3000
{{- end }}
